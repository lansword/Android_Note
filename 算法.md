## 1、快速排序

```java
public void fastSort(int[] a[],int left,int right){
  if(left >= right){
    return;
  }
  int p = a[left];
  int i = left,j = right;
  while(i < j){
    while(i < j && a[j] >= p){
      j--;
    }
    if(i < j){
      a[i++] = a[j];
    }
    while(i < j && a[i] <= p){
      i++;
    }
    if(i < j){
      a[j--] = a[i];
    }
  }
  a[i] = p;
  fastSort(a,left,i-1);
  fastSort(a,i+1,right);
}
```
